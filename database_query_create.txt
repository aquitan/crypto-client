
** base tables for mysql **


CREATE TABLE user_auth (ID INT AUTO_INCREMENT PRIMARY KEY, email VARCHAR(255) NOT NULL, password VARCHAR(50) NOT NULL, name VARCHAR(50), activationLink VARCHAR(255) NOT NULL, self_registration VARCHAR(30) NOT NULL, agreement BOOLEAN NOT NUll promocode VARCHAR(50) NOT NULL, domain_name VARCHAR(255) NOT NULL, date_of_entry VARCHAR(50) NOT NULL );

CREATE TABLE user_logs (ID INT AUTO_INCREMENT PRIMARY KEY, email VARCHAR(255) NOT NULL, ip_address VARCHAR(150) NOT NULL, request_city VARCHAR(100) NOT NULL, country_name VARCHAR(50) NOT NULL, location VARCHAR(150) NOT NULL, browser VARCHAR(50) NOT NULL, action_date VARCHAR(50) NOT NULL, user_action VARCHAR(255) NOT NULL, user_domain VARCHAR(255) NOT NULL, user_id INT NOT NULL, FOREIGN KEY(user_id) REFERENCES user_auth(ID));

CREATE TABLE user_params (ID INT AUTO_INCREMENT PRIMARY KEY, double_deposit BOOLEAN NOT NULL, swap_ban BOOLEAN NOT NULL, internal_ban BOOLEAN NOT NULL, isUser BOOLEAN NOT NULL, isStaff BOOLEAN NOT NULL, isAdmin BOOLEAN NOT NULL, isBanned BOOLEAN NOT NULL, isActivated BOOLEAN NOT NULL, premium_status BOOLEAN NOT NULL, two_step_status BOOLEAN NOT NULL, agreement BOOLEAN NOT NUll, user_id INT NOT NULL, FOREIGN KEY (user_id) REFERENCES user_auth(ID));

*** CREATE TABLE user_info_for_action (ID INT AUTO_INCREMENT PRIMARY KEY, deposit_fee INT NOT NULL, last_deposit DATE ,active_notification VARCHAR(1024) , active_error VARCHAR(1024) NOT NULL FOREIGN KEY  REFERENCES user_withdrawal_error, user_id INT NOT NULL, FOREIGN KEY (user_id) REFERENCES user_auth(ID) );


CREATE TABLE user_ip_match (ID INT AUTO_INCREMENT PRIMARY KEY, user_email VARCHAR(255) NOT NULL, user_ip_address VARCHAR(150) NOT NULL, login_date VARCHAR(50) NOT NULL, browser VARCHAR(50) NOT NULL );

CREATE TABLE user_kyc (ID INT AUTO_INCREMENT PRIMARY KEY, first_name VARCHAR(255) NOT NULL, last_name VARCHAR(255) NOT NULL, email VARCHAR(150) NOT NULL, phone_number VARCHAR(50) NOT NULL, date_of_birth VARCHAR(50) NOT NULL, document_number VARCHAR(100) NOT NULL, main_address VARCHAR(255) NOT NULL, sub_address VARCHAR(255), city VARCHAR(30) NOT NULL, country_name VARCHAR(50) NOT NULL, state VARCHAR(100) NOT NULL, zip_code INT NOT NULL, document_type VARCHAR(50) NOT NULL, kyc_status VARCHAR(30) NOT NULL, user_id INT NOT NULL, FOREIGN KEY(user_id) REFERENCES user_auth (ID) );

CREATE TABLE auth_token (ID INT AUTO_INCREMENT PRIMARY KEY, refresh_token VARCHAR(1024) NOT NULL, user_id INT NOT NULL, FOREIGN KEY(user_id) REFERENCES user_auth(ID));

CREATE TABLE user_promocode (ID INT AUTO_INCREMENT PRIMARY KEY, code VARCHAR(30) NOT NULL, date VARCHAR(50) NOT NULL, value FLOAT NOT NULL, staff_user_id INT NOT NULL, domain_name VARCHAR(255) NOT NULL, FOREIGN KEY (staff_user_id) REFERENCES user_auth(ID) );

CREATE TABLE staff_logs (ID INT AUTO_INCREMENT PRIMARY KEY, staff_email VARCHAR(255) NOT NULL, staff_action VARCHAR(320) NOT NULL, staff_domain VARCHAR(255) NOT NULL, staff_id INT NOT NULL, FOREIGN KEY (staff_id) REFERENCES user_auth(ID));

CREATE TABLE staff_params (ID INT AUTO_INCREMENT PRIMARY KEY, staff_email VARCHAR(255) NOT NULL, payment_fee INT NOT NULL, support_name VARCHAR(100) NOT NUll, get_staff_access_date VARCHAR(50) NOT NULL, user_who_give_access INT NOT NULL, FOREIGN KEY (user_who_give_access) REFERENCES user_auth(ID) );

CREATE TABLE used_promocode (ID INT AUTO_INCREMENT PRIMARY KEY, code VARCHAR(30) NOT NULL, date VARCHAR(50) NOT NULL, value FLOAT NOT NULL, staff_user_id INT NOT NULL, domain_name VARCHAR(255) NOT NULL, used_by_user VARCHAR(255) NOT NULL );

CREATE TABLE domain_list (ID INT AUTO_INCREMENT PRIMARY KEY, domain_name VARCHAR(100) NOT NULL, creation_date VARCHAR(100) NOT NULL, domain_owner INT NOT NULL, FOREIGN KEY(domain_owner) REFERENCES user_auth(ID));


CREATE TABLE user_balance (ID INT AUTO_INCREMENT PRIMARY KEY, coin_name VARCHAR(30) NOT NULL, coin_full_name VARCHAR(50) NOT NULL, coin_balance FLOAT NOT NULL, coin_rate FLOAT NOT NULL, user_id INT NOT NULL, FOREIGN KEY (user_id) REFERENCES user_auth(ID) );

CREATE TABLE user_wallet (ID INT AUTO_INCREMENT PRIMARY KEY, coin_name VARCHAR(30) NOT NULL, coin_full_name VARCHAR(50) NOT NULL, wallet_address VARCHAR(60), user_id INT NOT NULL, FOREIGN KEY (user_id) REFERENCES user_auth(ID) );

CREATE TABLE support_chat (ID INT AUTO_INCREMENT PRIMARY KEY, message_date VARCHAR(50) NOT NULL, message_body VARCHAR(550) NOT NULL, user_email VARCHAR(255) NOT NULL, staff_email VARCHAR(255) NOT NULL, staff_id INT NOT NULL, FOREIGN KEY (staff_id) REFERENCES user_auth(ID) );
